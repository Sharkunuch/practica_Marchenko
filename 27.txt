import datetime
def printTimeStamp(name):
    print('Автор програми: ' + name)
    print('Час компіляції: ' + str(datetime.datetime.now()))

time = [22, 23, 24, 1, 2, 3, 4, 5, 6, 7, 8]

papuga = input("+ Якщо папуга говорить, else - ")
my_time = int(input("Скільки часу? "))
if papuga == '-':
    print("Папуга мовчить")
elif papuga == '+' and my_time not in time:
    print("Зараз день")
else:
    print("Накрий клітку рядниною")
